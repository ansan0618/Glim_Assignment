#include <string>
#include <vector>
#include <iostream>

using namespace std;

int solution(vector<vector<int>> dots) {
    int answer = 0;
    vector<int> line1;
    int rad1;
    vector<int> line2;
    int rad2;
    for (int i = 0;i < 3;i++) {
        switch (i) {
        case 0:
            line1 = { dots[0][0] - dots[1][0], dots[0][1] - dots[1][1] };
            line2 = { dots[2][0] - dots[3][0], dots[2][1] - dots[3][1] };
            rad1 = line1[1] / line1[0];
            rad2 = line2[1] / line2[0];
            if (rad1 == rad2) {
                answer = 1;
            }
            break;
        case 1:
            line1 = { dots[0][0] - dots[2][0], dots[0][1] - dots[2][1] };
            line2 = { dots[1][0] - dots[3][0], dots[1][1] - dots[3][1] };
            rad1 = line1[1] / line1[0];
            rad2 = line2[1] / line2[0];
            if (rad1 == rad2) {
                answer = 1;
            }
            break;
        case 2:
            line1 = { dots[0][0] - dots[3][0], dots[0][1] - dots[3][1] };
            line2 = { dots[1][0] - dots[2][0], dots[1][1] - dots[2][1] };
            rad1 = line1[1] / line1[0];
            rad2 = line2[1] / line2[0];
            if (rad1 == rad2) {
                answer = 1;
            }
            break;
        }
    }
    return answer;
}


int main() {
    vector<vector<int>> dots1 = { {1, 4}, {9, 2}, {3, 8}, {11, 6} };
    vector<vector<int>> dots2 = { {3, 5},{4, 1},{2, 4},{5, 10} };
    int answer1 = solution(dots1);
    int answer2 = solution(dots2);
    std::cout << answer1 << std::endl;
    std::cout << answer2 << std::endl;
    
}
